60 #include <stdio.h>
61 #include <cinttypes>
62 
63 #define NN 312
64 #define MM 156
65 #define MATRIX_A 0xB5026F5AA96619E9ULL
66 #define UM 0xFFFFFFFF80000000ULL /* Most significant 33 bits */
67 #define LM 0x7FFFFFFFULL /* Least significant 31 bits */
68 
69 
70 /* The array for the state vector */
71 static uint64_t mt[NN];
72 /* mti==NN+1 means mt[NN] is not initialized */
73 static int mti=NN+1;
74 
75 /* initializes mt[NN] with a seed */
76 void init_genrand64(uint64_t seed)
77 {
78  mt[0] = seed;
79  for (mti=1; mti<NN; mti++)
80  mt[mti] = (6364136223846793005ULL * (mt[mti-1] ^ (mt[mti-1] >> 62)) + mti);
81 }
82 
83 /* initialize by an array with array-length */
84 /* init_key is the array for initializing keys */
85 /* key_length is its length */
86 void init_by_array64(uint64_t init_key[],
87  uint64_t key_length)
88 {
89  uint64_t i, j, k;
90  init_genrand64(19650218ULL);
91  i=1; j=0;
92  k = (NN>key_length ? NN : key_length);
93  for (; k; k--) {
94  mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >> 62)) * 3935559000370003845ULL))
95  + init_key[j] + j; /* non linear */
96  i++; j++;
97  if (i>=NN) { mt[0] = mt[NN-1]; i=1; }
98  if (j>=key_length) j=0;
99  }
100  for (k=NN-1; k; k--) {
101  mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >> 62)) * 2862933555777941757ULL))
102  - i; /* non linear */
103  i++;
104  if (i>=NN) { mt[0] = mt[NN-1]; i=1; }
105  }
106 
107  mt[0] = 1ULL << 63; /* MSB is 1; assuring non-zero initial array */
108 }

110 /* generates a random number on [0, 2^64-1]-interval */
111 uint64_t genrand64_int64(void)
112 {
113  int i;
114  uint64_t x;
115  static uint64_t mag01[2]={0ULL, MATRIX_A};
116 
117  if (mti >= NN) { /* generate NN words at one time */
118 
119  /* if init_genrand64() has not been called, */
120  /* a default initial seed is used */
121  if (mti == NN+1)
122  init_genrand64(5489ULL);
123 
124  for (i=0;i<NN-MM;i++) {
125  x = (mt[i]&UM)|(mt[i+1]&LM);
126  mt[i] = mt[i+MM] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
127  }
128  for (;i<NN-1;i++) {
129  x = (mt[i]&UM)|(mt[i+1]&LM);
130  mt[i] = mt[i+(MM-NN)] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
131  }
132  x = (mt[NN-1]&UM)|(mt[0]&LM);
133  mt[NN-1] = mt[MM-1] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
134 
135  mti = 0;
136  }
137 
138  x = mt[mti++];
139 
140  x ^= (x >> 29) & 0x5555555555555555ULL;
141  x ^= (x << 17) & 0x71D67FFFEDA60000ULL;
142  x ^= (x << 37) & 0xFFF7EEE000000000ULL;
143  x ^= (x >> 43);
144 
145  return x;
146 }

149 long long genrand64_int63(void)

151  return (long long)(genrand64_int64() >> 1);

154 /* generates a random number on [0,1]-real-interval */
155 double genrand64_real1(void)
156 {
157  return (genrand64_int64() >> 11) * (1.0/9007199254740991.0);
158 }
